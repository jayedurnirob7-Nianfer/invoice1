    var dataStart=looksLikeHeaders?2:1;
    sheetRows=[];
    for(var r=dataStart;r<rawRows.length;r++){
      if(!rawRows[r]) continue;
      var vals=rawRows[r]; var row={};
      for(var c=1;c<vals.length;c++){ var h=headers[c-1]||('col'+c); row[h]=(vals[c]||'').trim(); }
      if(Object.values(row).some(function(v){ return v; })) sheetRows.push(row);
    }
    renderTable();
  }

  /* ── Parse CSV ── */
  function parsePastedCSV(){ var text=g('csv-paste').value.trim(); if(!text){ toast('Paste some CSV data first'); return; } parseCSV(text); if(sheetRows.length) toast('✓ Loaded '+sheetRows.length+' rows'); }

  var POSITIONAL_HEADERS=['ShippingNumber','TrackingNumber','ShippingMark','Date','Name','Address','Mobile','Description','Quantity','Weight','Cost','Total'];

  function parseCSV(text){
    text=text.replace(/\r\n/g,'\n').replace(/\r/g,'\n');
    var lines=text.trim().split('\n'); if(!lines.length){ toast('No data found'); return; }
    var firstLine=splitLine(lines[0]).map(function(h){ return h.replace(/^"|"$/g,'').trim(); });
    var allBlank=firstLine.every(function(h){ return h===''; });
    var headers=allBlank?POSITIONAL_HEADERS:firstLine;
    var dataLines=allBlank?lines:lines.slice(1);
    sheetRows=dataLines.filter(function(l){ return l.trim(); }).map(function(line){
      var vals=splitLine(line); var row={};
      headers.forEach(function(h,i){ if(h) row[h]=(vals[i]||'').replace(/^"|"$/g,'').trim(); });
      return row;
    });
    renderTable();
  }

  function splitLine(line){
    var result=[],cur='',inQ=false;
    for(var i=0;i<line.length;i++){
      var ch=line[i];
      if(ch==='"'){ if(inQ&&line[i+1]==='"'){ cur+='"'; i++; } else { inQ=!inQ; } }
      else if(ch===','&&!inQ){ result.push(cur); cur=''; }
      else { cur+=ch; }
    }
    result.push(cur); return result;
  }

  /* ── Manual row ── */
  function addManualRow(){
    var row={}; var fields=document.querySelectorAll('.mf'); var hasData=false;
    fields.forEach(function(f){ var val=f.value.trim(); row[f.getAttribute('data-key')]=val; if(val) hasData=true; });
    if(!hasData){ toast('Fill in at least one field'); return; }
    if(!row['Name']){ toast('Name is required'); return; }
    sheetRows.push(row); renderTable();
    fields.forEach(function(f){ f.value=''; });
    toast('✓ Row added ('+sheetRows.length+' total)');
  }

  /* ── Detect columns ── */
  function detectCols(row){
    var keys=Object.keys(row);
    var strip=function(s){ return String(s).toLowerCase().replace(/[\s_\-\.]/g,''); };
    function find(){
      var names=Array.prototype.slice.call(arguments);
      var found=keys.find(function(k){ return names.some(function(n){ return strip(k)===strip(n); }); });
      if(found) return found;
      found=keys.find(function(k){ return names.some(function(n){ return strip(k).indexOf(strip(n))>=0; }); });
      if(found) return found;
      found=keys.find(function(k){ return names.some(function(n){ return strip(n).indexOf(strip(k))>=0; }); });
      return found||null;
    }
    return {
      Name:find('Name','client','customer','fullname'),
      Address:find('Address','addr','location','area'),
      Mobile:find('Mobile','phone','tel','contact'),
      ShippingMark:find('ShippingMark','Shipping Mark','mark','shipmark'),
      ShippingNumber:find('ShippingNumber','Shipping Number','shippingno','shipno'),
      TrackingNumber:find('TrackingNumber','Tracking Number','trackingno','trackno'),
      Description:find('Description','Product Name','desc','product','item','goods'),
      Weight:find('Weight','wt','kg','weights'),
      Unit:find('Quantity','Pcs','pcs','Unit','qty','quantity'),
      Cost:find('Cost','Rate','rate','price','costkg'),
      Total:find('Total','amount','sum','totalcost'),
      Date:find('Date','dt','shipdate','invoicedate'),
    };
  }

  /* ── Render Table ── */

  }

  /* ── Selection ── */
  function updateSel(){
    selectedRows=[];
    document.querySelectorAll('.row-chk').forEach(function(chk){
      var idx=parseInt(chk.getAttribute('data-idx'));
      var tr=chk.closest?chk.closest('tr'):chk.parentNode.parentNode;
      if(chk.checked){ selectedRows.push(sheetRows[idx]); tr.classList.add('selected'); }
      else { tr.classList.remove('selected'); }
    });
    var badge=g('sel-badge');
    badge.style.display=selectedRows.length?'inline-block':'none';
    badge.textContent=selectedRows.length+' selected';
  }
  function toggleAll(checked){ document.querySelectorAll('.row-chk').forEach(function(c){ c.checked=checked; }); updateSel(); }
  function selectAll(){ g('check-all').checked=true; toggleAll(true); }
  function clearSel(){ g('check-all').checked=false; toggleAll(false); }

  /* ── Generate Invoice ── */
  function generateInvoice(){
    if(!selectedRows.length){ toast('Select at least one row first'); return; }
    var col=detectCols(sheetRows[0]);
    var first=selectedRows[0];
    setText('inv-name', first[col.Name]);
    setText('inv-address', first[col.Address]);
    setText('inv-mobile', first[col.Mobile]);
    setText('inv-ship-mark', first[col.ShippingMark]);
    setText('inv-ship-no', first[col.ShippingNumber]);
    setText('inv-track-no', first[col.TrackingNumber]);
    var dateVal=first[col.Date];
    if(!dateVal||dateVal==='—') dateVal=new Date().toLocaleDateString('en-GB');
    setText('inv-date', dateVal);
    var grand=0, html='';
    for(var i=0;i<selectedRows.length;i++){
      var r=selectedRows[i];
      var weight=parseFloat(String(r[col.Weight]||'0').replace(/[^0-9.]/g,''))||0;
      var cost=parseFloat(String(r[col.Cost]||'0').replace(/[^0-9.]/g,''))||0;
      var rawTot=parseFloat(String(r[col.Total]||'0').replace(/[^0-9.]/g,''))||0;
      var lineTotal=(weight&&cost)?(weight*cost):rawTot;
      grand+=lineTotal;
      html+='<tr>'
        +'<td>'+String(i+1).padStart(2,'0')+'</td>'
        +'<td>'+esc(r[col.Description])+'</td>'
        +'<td>'+esc(r[col.Weight])+' '+esc(r[col.Unit])+'</td>'
        +'<td>X</td>'
        +'<td>'+esc(r[col.Cost])+' / KG</td>'
        +'<td>'+(lineTotal?lineTotal.toLocaleString('en-IN',{minimumFractionDigits:2}):'—')+'</td>'
        +'</tr>';
    }
    g('inv-products').innerHTML=html;
    setText('inv-grand-total', grand?grand.toLocaleString('en-IN',{minimumFractionDigits:2}):'0.00');
    g('invoice-modal').classList.add('open');
  }

  function closeModal(){ g('invoice-modal').classList.remove('open'); }
